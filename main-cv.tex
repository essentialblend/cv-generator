\newcommand{\shirleyTracerProject}
{
	\textbf{Shirley-Tracer}%
	\hfill \href{https://siddharths.space/projects/shirley-tracer}{siddharths.space/projects/shirley-tracer} \\
	\textit{A C++ Monte-Carlo path tracer inspired by Peter Shirley's "Ray Tracing in One Weekend" series}\\[-4pt]
	\begin{itemize}
		\item Implemented a multi-threaded rendering architecture and key variance reduction techniques including Multiple Importance Sampling (MIS) to combine light and material BSDF sampling strategies
		\item Accelerated ray-scene intersections with a Bounding Volume Hierarchy (AA-BVH), improving performance
		\item Developed a versatile material system with metal, dielectrics, and diffuse surfaces with participating media
		\item Built a positionable camera model supporting adjustable defocus blur and motion blur.
	\end{itemize}
}

\newcommand{\indusProject}
{
	\textbf{Indus}%
	\hfill \href{https://github.com/essentialblend/project-indus/tree/rewrite\_cpp23}{siddharths.space/projects/project-indus}\\
	\textit{Multithreaded PBRT-inspired CPU path tracer in modern C++23 and SFML\,---\,(ongoing development)}\\[-4pt]
	\begin{itemize}
		\item Built a modular C++23 CPU path tracer using named modules, concepts-constrained geometry/math types, and polymorphic renderer subsystems
		\item Implemented numerical robustness and acceleration systems: FMA utilities, interval arithmetic, ULP-offset secondary rays, and a flattened 32-bucket SAH BVH
		\item Implemented an environment-lit multi-bounce BSDF path integrator with PCG32/stratified sampling, Lambertian/dielectric transport, Fresnel evaluation, and Russian roulette
		\item Added thin-lens depth of field, object motion blur, RGB film accumulation, Gaussian filtering, PNG output, and \texttt{std::jthread} tile scheduling with progressive SFML preview/HUD
	\end{itemize}

}
